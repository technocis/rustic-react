# ListItem

This is the `RusticUI` ListItem Component. It is fairly simple. You just wrap whatever content you want inside of `<ListItem>` and `</ListItem>`.

import { Canvas, Meta, Story, ArgsTable } from "@storybook/addon-docs";
import { ListItem, NestedList } from "./ListItem";
import { FiHome, FiGrid } from "react-icons/fi";

<Meta title="List Item" component={ListItem} />

export const Template = (args) => (
  <ul>
    <ListItem text="Item 1" subtext="This is subtext for Item 1"></ListItem>
    <ListItem text="Item 2" active={true}></ListItem>
  </ul>
);

<Canvas>
  <Story name="List Item">{Template.bind({})}</Story>
</Canvas>
